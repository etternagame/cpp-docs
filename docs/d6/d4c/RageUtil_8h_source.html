892 struct char_traits_char_nocase : public char_traits<char>
893 {

895  {

897  g_UpperCase[static_cast<unsigned char>(c2)];
898  }
899 
900  static bool ne(char c1, char c2)
901  {
902  return g_UpperCase[static_cast<unsigned char>(c1)] !=
903  g_UpperCase[static_cast<unsigned char>(c2)];
904  }
905 
906  static bool lt(char c1, char c2)
907  {
908  return g_UpperCase[static_cast<unsigned char>(c1)] <
909  g_UpperCase[static_cast<unsigned char>(c2)];
910  }
911 
912  static int compare(const char* s1, const char* s2, size_t n)
913  {
914  int ret = 0;
915  while (n--) {
916  ret = fasttoupper(*s1++) - fasttoupper(*s2++);
917  if (ret != 0)
918  break;
919  }
920  return ret;
921  }
922 
923  static char fasttoupper(char a)
924  {
925  return g_UpperCase[static_cast<unsigned char>(a)];
926  }
927 
928  static const char* find(const char* s, int n, char a)
929  {
930  a = fasttoupper(a);
931  while (n-- > 0 && fasttoupper(*s) != a)
932  ++s;
933 
934  if (fasttoupper(*s) == a)
935  return s;
936  return nullptr;
937  }
938 };
