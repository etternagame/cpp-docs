44 splitWorkLoad(vector<T>& v, size_t elementsPerThread)
45 {
46  std::vector<vectorRange<T>> ranges;
47  if (elementsPerThread <= 0 || elementsPerThread >= v.size()) {
48  ranges.push_back(std::make_pair(v.begin(), v.end()));
49  return ranges;
50  }
51 
52  size_t range_count = (v.size() + 1) / elementsPerThread + 1;
53  size_t ePT = v.size() / range_count;
54  if (ePT == 0) {
55  ranges.push_back(std::make_pair(v.begin(), v.end()));
56  return ranges;
57  }
58  size_t i;
59 
60  vectorIt<T> b = v.begin();
61 
62  for (i = 0; i < v.size() - ePT; i += ePT)
63  ranges.push_back(std::make_pair(b + i, b + i + ePT));
64 
65  ranges.push_back(std::make_pair(b + i, v.end()));
66  return ranges;
67 }

69 template<typename T>
70 void
71 parallelExecution(vector<T> vec,
72  function<void(int)> update,
73  function<void(vectorRange<T>, ThreadData*)> exec,
74  void* stuff)
75 {
76  const int THREADS = std::thread::hardware_concurrency();
77  std::vector<vectorRange<T>> workloads =
78  splitWorkLoad(vec, static_cast<size_t>(vec.size() / THREADS));
79  ThreadData data;
80  data.data = stuff;
81  auto threadCallback = [&data, &exec](vectorRange<T> workload) {
82  exec(workload, &data);
83  data._threadsFinished++;
84  data.setUpdated(true);
85  };
86  vector<thread> threadpool;
87  for (auto& workload : workloads)
88  threadpool.emplace_back(thread(threadCallback, workload));
89  while (data._threadsFinished < workloads.size()) {
90  data.waitForUpdate();
91  update(data._progress);
92  data.setUpdated(false);
93  }
94  for (auto& thread : threadpool)
95  thread.join();
96 }
