601 // RageMatrix elements are specified in row-major order. This
602 // means that the translate terms are located in the fourth row and the
603 // projection terms in the fourth column. This is consistent with the way
604 // MAX, Direct3D, and OpenGL all handle matrices. Even though the OpenGL
605 // documentation is in column-major form, the OpenGL code is designed to
606 // handle matrix operations in row-major form.
607 struct RageMatrix

609  public:
610  RageMatrix() = default;
611  RageMatrix(const float* f)
612  {
613  for (int i = 0; i < 4; i++)
614  for (int j = 0; j < 4; j++)
615  m[j][i] = f[j * 4 + i];
616  }

618  {
619  for (int i = 0; i < 4; i++)

621  m[j][i] = other.m[j][i];
