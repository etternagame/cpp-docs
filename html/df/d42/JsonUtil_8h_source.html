306 static void
307 DeserializeObjectToObjectMapAsArray(map<K, V>& m,
308  const RString& sKeyName,
309  const RString& sValueName,
310  const Json::Value& root)
311 {
312  m.clear();
313  ASSERT(root.type() == Json::arrayValue);
314  for (Json::Value::const_iterator iter = root.begin(); iter != root.end();
315  iter++) {
316  ASSERT((*iter).type() == Json::objectValue);
317  K k;
318  if (!k.Deserialize((*iter)[sKeyName]))
319  continue;
320  V v;
321  if (!v.Deserialize((*iter)[sValueName]))
322  continue;
323  m[k] = v;
324  }
325 }
