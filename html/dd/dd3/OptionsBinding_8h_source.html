10 // Functions are designed to combine Get and Set into one, to be less clumsy to
11 // use. -Kyz If a valid arg is passed, the value is set. The previous value is
12 // returned. OPTIONAL_RETURN_SELF exists to provide optional chaining support.
13 // Example: local a= player_options:Twirl(5, 1, true):Roll(5,
14 // true):Dizzy(true):Twirl()
15 #define OPTIONAL_RETURN_SELF(option_index) \
16  if (lua_isboolean(L, option_index) && lua_toboolean(L, option_index)) { \
17  p->PushSelf(L); \
18  return 1; \
19  }

95  { \
96  int original_top = lua_gettop(L); \
97  lua_createtable(L, p->m_##member.size(), 0); \
98  for (size_t n = 0; n < p->m_##member.size(); ++n) { \
99  lua_pushnumber(L, p->m_##member[n]); \
100  lua_rawseti(L, -2, n + 1); \
101  } \
102  if (lua_istable(L, 1) && original_top >= 1) { \
103  size_t size = lua_objlen(L, 1); \
104  if (valid(L, 1)) { \
