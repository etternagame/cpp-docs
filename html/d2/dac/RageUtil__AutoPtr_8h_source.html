6 /*
7  * This is a simple copy-on-write refcounted smart pointer. Once constructed,
8  * all read-only access to the object is made without extra copying. If you
9  * need read-write access, you can get a pointer with Get(), which will cause
10  * the object to deep-copy. (Don't free the resulting pointer.)
11  *
12  * Note that there are no non-const operator* or operator-> overloads, because
13  * that would cause all const access by code with non-const permissions to
14  * deep-copy. For example,
15  *
16  * AutoPtrCopyOnWrite<int> a( new int(1) );
17  * AutoPtrCopyOnWrite<int> b( a );
18  * printf( "%i\n", *a );
19  *
20  * If we have a non-const operator*, this *a will use it (even though it only
21  * needs const access), and will copy the underlying object wastefully. g++
22  * std::string has this behavior, which is why it's important to qualify strings
23  * as "const" when const access is desired, but that's brittle, so let's make
24  * all potential deep-copying explicit.
25  */

104 /*
105  * This smart pointer template is used to safely hide implementations from
106  * headers, to reduce dependencies. This is the same as declaring a pointer
107  * to a class, and allocating/deallocating it in the implementation: only
108  * the implementation needs to include that class. This makes copying
109  * and deletion automatic, so you don't need to include a copy ctor or
110  * remember to delete it.
111  *
112  * There's one subtlety: in order to copy or delete an object, we need its
113  * definition. This is intended to avoid pulling in the definition. So,
114  * we use a traits class to hide it. Use REGISTER_CLASS_TRAITS for each
115  * class used with this template.
116  *
117  * Concepts from http://www.gotw.ca/gotw/062.htm.
118  */
119 template<class T>
120 struct HiddenPtrTraits
121 {
122  static T* Copy(const T* pCopy);
123  static void Delete(T* p);
124 };
125 #define REGISTER_CLASS_TRAITS(T, CopyExpr) \
126  template<> \
127  T* HiddenPtrTraits<T>::Copy(const T* pCopy) \
128  { \
129  return CopyExpr; \
130  } \
131  template<> \
132  void HiddenPtrTraits<T>::Delete(T* p) \
133  { \
134  delete p; \
135  }
