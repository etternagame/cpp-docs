8 /* Lock-free circular buffer. This should be threadsafe if one thread is

15  * data is written to. If read_pos == write_pos, the buffer is empty.
16  *
17  * There will always be at least one position empty, as a completely full
18  * buffer (read_pos == write_pos) is indistinguishable from an empty buffer.

20  * Invariants: read_pos < size, write_pos < size. */

136  void get_write_pointers(T* pPointers[2], unsigned pSizes[2])
137  {
138  const int rpos = read_pos;
139  const int wpos = write_pos;
140 
141  if (rpos <= wpos) {
142  /* The buffer looks like "eeeeDDDDeeee" or "eeeeeeeeeeee" (e =
143  * empty, D = data). */
144  pPointers[0] = buf + wpos;
145  pPointers[1] = buf;
146 
147  pSizes[0] = size - wpos;
148  pSizes[1] = rpos;
149  } else if (rpos > wpos) {
150  /* The buffer looks like "DDeeeeeeeeDD" (e = empty, D = data). */
151  pPointers[0] = buf + wpos;
152  pPointers[1] = NULL;
153 
154  pSizes[0] = rpos - wpos;
155  pSizes[1] = 0;
156  }

159  * while keeping the entries aligned to m_iBlockSize. */
