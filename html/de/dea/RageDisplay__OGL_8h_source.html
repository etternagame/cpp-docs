10 /* Making an OpenGL call doesn't also flush the error state; if we happen
11  * to have an error from a previous call, then the assert below will fail.
12  * Flush it. */
13 #define FlushGLErrors() \
14  do { \
15  } while (glGetError() != GL_NO_ERROR)
16 #define AssertNoGLError() \
17  { \
18  GLenum error = glGetError(); \
19  ASSERT_M(error == GL_NO_ERROR, \
20  RageDisplay_Legacy_Helpers::GLToString(error)); \
21  }
