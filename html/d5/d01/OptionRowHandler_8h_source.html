258 inline void
259 VerifySelected(SelectType st, vector<bool>& selected, const RString& sName)
260 {
261  int num_selected = 0;
262  if (st == SELECT_ONE) {
263  int first_selected = -1;
264  if (selected.empty()) {
265  LuaHelpers::ReportScriptErrorFmt(
266  "Option row %s requires only one "
267  "thing to be selected, but the list of selected things has zero "
268  "elements.",
269  sName.c_str());
270  return;
271  }
272  for (unsigned int e = 0; e < selected.size(); ++e) {
273  if (selected[e]) {
274  num_selected++;
275  if (first_selected == -1) {
276  first_selected = static_cast<int>(e);
277  }
278  }
279  }
280  if (num_selected != 1) {
281  LuaHelpers::ReportScriptErrorFmt(
282  "Option row %s requires only one "
283  "thing to be selected, but %i out of %i things are selected.",
284  sName.c_str(),
285  num_selected,
286  static_cast<int>(selected.size()));
287  for (unsigned int e = 0; e < selected.size(); ++e) {
288  if (selected[e] && e != first_selected) {
289  selected[e] = false;
290  }
291  }
292  if (num_selected == 0) {
293  selected[0] = true;
294  }
295  return;
296  }
297  }
