109  static int thunk(Lua* L)
110  {
111  // stack has userdata, followed by method args
112  T* obj = check(L, 1, true); // get self
113  lua_remove(L,
114  1); // remove self so member function args start at index 1
115  // get member function from upvalue
116  binding_t* pFunc = (binding_t*)lua_touserdata(L, lua_upvalueindex(1));
